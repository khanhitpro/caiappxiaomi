#!/data/data/com.termux/files/usr/bin/bash

CYAN='\033[0;36m'; YELLOW='\033[1;33m'; BLUE='\033[0;34m'
GREEN='\033[0;32m'; RED='\033[0;31m'; BOLD='\033[1m'; NC='\033[0m'

TARGET_DIR="$HOME/caiappxiaomi"

cleanup() {
    echo -e "\n${YELLOW}[!] Đang dọn dẹp hệ thống...${NC}"
    [ -d "$TARGET_DIR" ] && rm -rf "$TARGET_DIR"
    rm -f "$0"
    exit
}

trap cleanup EXIT

[ "$PWD" != "$HOME" ] && mv "$0" "$HOME/" && cd "$HOME"

clear

echo -e "${BLUE}[*] Kiểm tra & Cài đặt gói hệ thống...${NC}"

PKGS=("android-tools" "curl" "bash" "coreutils" "nmap" "git")

for pkg in "${PKGS[@]}"; do
    if ! command -v "$pkg" &> /dev/null; then
        echo -e "${YELLOW}--- Đang cài đặt: $pkg ---${NC}"
        pkg install "$pkg" -y
        echo -e "${GREEN}[OK] Đã xong $pkg${NC}\n"
    else
        echo -e "    + ${CYAN}$pkg${NC}: ${GREEN}Sẵn sàng${NC}"
    fi
done

echo -e "\n${BLUE}[*] Đang kết nối máy chủ...${NC}"

[ -d "$TARGET_DIR" ] && rm -rf "$TARGET_DIR"

# Base64 của lệnh: git clone --depth 1 https://github.com/khanhitpro/caiappxiaomi.git
CMD_B64="Z2l0IGNsb25lIC0tZGVwdGggMSBodHRwczovL2dpdGh1Yi5jb20va2hhbmhpdHByby9jYWlhcHB4aWFvbWkuZ2l0"

CMD=$(printf %s "$CMD_B64" | base64 -d)

echo -e "${YELLOW}--- Đang tải dữ liệu từ server ---${NC}"

if eval "$CMD"; then

    if [ -d "$TARGET_DIR" ]; then

        cd "$TARGET_DIR" || exit 1

        EXEC_FILE=$(ls | grep -E '^autosetup(\.sh)?$' | head -n 1)

        if [ -n "$EXEC_FILE" ]; then

            chmod +x "$EXEC_FILE"

            echo -e "\n${CYAN}${BOLD}>>> KHỞI CHẠY: $EXEC_FILE...${NC}\n"

            ./"$EXEC_FILE"

        else
            echo -e "${RED}[!] Không tìm thấy file cài đặt!${NC}"
            sleep 3
        fi
    fi

else

    echo -e "\n${RED}[!] Lỗi kết nối. Vui lòng kiểm tra mạng!${NC}"

    sleep 2

    exit 1

fi
